## Quick Start

### Create Group with Members

```hcl
# Create group
resource "microsoft365_graph_beta_groups_group" "engineering" {
  display_name     = "Engineering Team"
  mail_nickname    = "engineering"
  description      = "Engineering department group"
  security_enabled = true
  mail_enabled     = false
}

# Add members
resource "microsoft365_graph_beta_groups_group_member_assignment" "member1" {
  group_id  = microsoft365_graph_beta_groups_group.engineering.id
  member_id = microsoft365_graph_beta_users_user.engineer1.id
}

# Add owner
resource "microsoft365_graph_beta_groups_group_owner_assignment" "owner" {
  group_id = microsoft365_graph_beta_groups_group.engineering.id
  owner_id = microsoft365_graph_beta_users_user.manager.id
}
```
